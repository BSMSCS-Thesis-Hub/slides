% Slide 8: Model Compression (Descriptive Placeholder)
\begin{frame}{Model Compression}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \textbf{Compression Gaps}
        \begin{itemize}
            \item Traditional quantization and pruning techniques significantly reduce client-side overhead.
            \item Over-compression destroys life-critical local movement features on low-resource wearables.
            \item Active dynamic feedback loops are necessary to adapt to statistical client data variance.
        \end{itemize}

% Slide 9: Federated Distillation (Descriptive Placeholder)
\begin{frame}{Federated Distillation}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \textbf{Distillation Mechanics}
        \begin{itemize}
            \item Client nodes exchange soft-label logits rather than raw network weights.
            \item Transmission sizes on communication channels are drastically minimized.
            \item Logit aggregation naturally mitigates label skew under non-IID conditions.
        \end{itemize}

        \column{0.55\textwidth}
        \centering

% Slide 10: Paradigm Comparison (New Visual Slide)
\begin{frame}{Paradigm Comparison}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \textbf{Federated Distillation Taxonomy}
        \begin{itemize}
            \item Feature-based FD shares only model outputs, significantly reducing communication overhead.
            \item Parameter-based FD uploads student networks, addressing model fusion constraints.
            \item Data-based FD exchanges synthetic datasets, lowering local compute costs.
        \end{itemize}
